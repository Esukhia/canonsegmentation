 འོད་ དེས་ སྟོང་གསུམ་ གྱི་ སྟོང་ ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ འདི་ ན་ །  སེམས་ཅན་ གང་ ཇི་སྙེད་ ཅིག་ ཡོད་པ་ དེ་དག་ ཐམས་ཅད་ ཀྱིས་ །  བྱང་ཕྱོགས་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གངྒཱའི་ ཀླུང་ གི་ བྱེ་མ་ སྙེད་ ཀྱི་ །  དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ པ་ ཡང་དག་ པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ ཉན་ཐོས་ ཀྱི་ དགེ་འདུན་ དང་ བཅས་པ་ །  བྱང་ཆུབ་སེམས་དཔའ་ ￥འི་ འཁོར་ དང་ བཅས་པ་ མང་པོ་ མཐོང་བ་ #ར་ གྱུར་ ཏོ་ ། ། བྱང་ཕྱོགས་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གངྒཱའི་ ཀླུང་ གི་ བྱེ་མ་ སྙེད་ པོ་ དེ་དག་ ན་ །  སེམས་ཅན་ གང་ ཇི་སྙེད་ འཁོད་པ་ དེ་དག་ གིས་ ཀྱང་ །  སྟོང་གསུམ་ གྱི་ སྟོང་ ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ འདི་ ན་ །  བཅོམ་ལྡན་འདས་ དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ པ་ ཡང་དག་ པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ ཤཱཀྱ་ ཐུབ་པ་ །  དགེ་སློང་ གི་ དགེ་འདུན་ དང་ བཅས་པ་ །  བྱང་ཆུབ་སེམས་དཔའ་ ￥འི་ འཁོར་ དང་ ཐབས་ གཅིག་ ཏུ་ བཞུགས་པ་ མཐོང་བ་ #ར་ གྱུར་ ཏོ་ ། ། འོད་ དེས་ སྟོང་གསུམ་ གྱི་ །  སྟོང་ ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ འདི་ ན་ །  སེམས་ཅན་ གང་ ཇི་སྙེད་ ཅིག་ ཡོད་པ་ དེ་དག་ ཐམས་ཅད་ ཀྱིས་ །  བྱང་ཤར་ གྱི་ ཕྱོགས་མཚམས་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གངྒཱའི་ ཀླུང་ གི་ བྱེ་མ་ སྙེད་ ཀྱི་ །  དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ པ་ ཡང་དག་ པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ ཉན་ཐོས་ ཀྱི་ དགེ་འདུན་ དང་ བཅས་པ་ །  བྱང་ཆུབ་སེམས་དཔའ་ ￥འི་ འཁོར་ དང་ བཅས་པ་ མང་པོ་ མཐོང་བ་ #ར་ གྱུར་ ཏོ་ ། ། བྱང་ཤར་ གྱི་ ཕྱོགས་མཚམས་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གངྒཱའི་ ཀླུང་ གི་ བྱེ་མ་ སྙེད་ པོ་ དེ་དག་ ན་ །  སེམས་ཅན་ གང་ ཇི་སྙེད་ འཁོད་པ་ དེ་དག་ གིས་ ཀྱང་ །  སྟོང་གསུམ་ གྱི་ སྟོང་ ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ འདི་ ན་ །  བཅོམ་ལྡན་འདས་ དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ པ་ ཡང་དག་ པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ ཤཱཀྱ་ ཐུབ་པ་ །  དགེ་སློང་ གི་ དགེ་འདུན་ དང་ བཅས་པ་ བྱང་ཆུབ་སེམས་དཔའ་ ￥འི་ འཁོར་ དང་ ཐབས་ གཅིག་ ཏུ་ བཞུགས་པ་ མཐོང་བ་ #ར་ གྱུར་ ཏོ་ ། །

 ཤར་ལྷོ་ #འི་ ཕྱོགས་མཚམས་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གངྒཱའི་ ཀླུང་ གི་ བྱེ་མ་ སྙེད་ པོ་ དེ་དེ་ དག་ ན་ །  སེམས་ཅན་ གང་ ཇི་སྙེད་ འཁོད་པ་ དེ་དག་ གིས་ ཀྱང་ །  སྟོང་གསུམ་ གྱི་ སྟོང་ ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ འདི་ ན་ །  བཅོམ་ལྡན་འདས་ དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ པ་ ཡང་དག་ པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ ཤཱཀྱ་ ཐུབ་པ་ །  དགེ་སློང་ གི་ དགེ་འདུན་ དང་ བཅས་པ་ །  བྱང་ཆུབ་སེམས་དཔའ་ ￥འི་ འཁོར་ དང་ ཐབས་ གཅིག་ ཏུ་ བཞུགས་པ་ མཐོང་བ་ #ར་ གྱུར་ ཏོ་ ། ། འོད་ དེས་ སྟོང་གསུམ་ གྱི་ སྟོང་ ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ འདི་ ན་ །  སེམས་ཅན་ གང་ ཇི་སྙེད་ ཅིག་ ཡོད་པ་ དེ་དག་ ཐམས་ཅད་ ཀྱིས་ །  ལྷོ་ནུབ་ ཀྱི་ ཕྱོགས་མཚམས་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གངྒཱའི་ ཀླུང་ གི་ བྱེ་མ་ སྙེད་ ཀྱི་ དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ པ་ ཡང་དག་ པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ ཉན་ཐོས་ ཀྱི་ དགེ་འདུན་ དང་ བཅས་པ་ །  བྱང་ཆུབ་སེམས་དཔའ་ ￥འི་ འཁོར་ དང་ བཅས་པ་ མང་པོ་ མཐོང་བ་ #ར་ གྱུར་ ཏོ་ ། ། ལྷོ་ནུབ་ ཀྱི་ ཕྱོགས་མཚམས་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གངྒཱའིཀླུང་ གི་ བྱེ་མ་ སྙེད་ པོ་ དེ་དག་ ན་ །  སེམས་ཅན་ གང་ ཇི་སྙེད་ འཁོད་པ་ དེ་དག་ གིས་ ཀྱང་ །  སྟོང་གསུམ་ གྱི་ སྟོང་ ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ འདི་ ན་ །  བཅོམ་ལྡན་འདས་ དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ པ་ ཡང་དག་ པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ ཤཱཀྱ་ ཐུབ་པ་ །  དགེ་སློང་ གི་ དགེ་འདུན་ དང་ བཅས་པ་ །  བྱང་ཆུབ་སེམས་དཔའ་ ￥འི་ འཁོར་ དང་ ཐབས་ གཅིག་ ཏུ་ བཞུགས་པ་ མཐོང་བ་ #ར་ གྱུར་ ཏོ་ ། །
