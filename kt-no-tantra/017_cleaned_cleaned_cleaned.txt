 ༄༅།  ཤེས་རབ་ ཀྱི་ ཕ་ རོལད་ ཏུ་ ཕྱིནད་ -པ་ སྟོང་ཕྲག་ བརྒྱ་ -པ་ །  གླེགས་བམ་ བཞི་པ་ །  བམ་པོ་ དང་པོ་ ། ། ཚེ་ -དང་ -ལྡན་པ་ ཤ་ར་དྭ་ཏི་ #འི་ བུ་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྡོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ སྟོང་པ་ཉིད་ ཀྱི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ དབེན་པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། 
 དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ སྟོང་པ་ཉིད་ ཀྱི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། 
 དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ དབེན་པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ དབུས་ སུ་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ སྟོང་པ་ཉིད་ ཀྱི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ དབུས་ སུ་ མྱི་ དམྱིགས་ -སོ་ ། ། དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ དབེན་པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ དབུས་ སུ་ མྱི་ དམྱིགས་ -སོ་ ། ། 
 དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ དབུས་ སུ་ མྱི་ དམྱིགས་ -སོ་ །  དེ་ ཅིའི་ཕྱིར་ ཞེ་ན་ །  ཚེ་ -དང་ -ལྡན་པ་ ཤ་ར་དྭ་ཏི་ #འི་ བུ་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ མྱེད་ -པ་ དང་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ སྟོང་པ་ཉིད་ དང་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ དབེན་པ་ དང་ །  དྲན་པ་ ཉེ་ པར་ གཞག་ -པ་ ངོ་བོ་ -ཉིད་ མྱེད་ པ་-ལ་ སྔོན་ གྱི་ མཐའ་ མྱི་ དམྱིགས་ ཕྱི་མ་ #འི་ མཐའ་ མྱི་ དམྱིགས་ །  དབུས་ མྱི་ དམྱིགས་ ཏེ་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ མྱེད་ -པ་ ཡང་ གཞན་མ་ ཡིན་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ སྟོང་པ་ཉིད་ ཀྱང་ གཞན་མ་ ཡིན་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ དབེན་པ་ ཡང་ གཞན་མ་ ཡིན་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ ཡང་ གཞན་མ་ ཡིན་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཡང་ གཞན་མ་ ཡིན་ །  སྔོན་ གྱི་ མཐའ་ ཡང་ གཞན་མ་ ཡིན་ །  ཕྱི་མ་ #འི་ མཐའ་ ཡང་ གཞན་མ་ ཡིན་ །  དབུས་ ཀྱང་ གཞན་མ་ ཡིན་ -ནོ་ ། ། ཚེ་ -དང་ -ལྡན་པ་ ཤ་ར་དྭ་ཏི་ #འི་ བུ་ །  དེ་ལྟར་ ན་ དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ མྱེད་ -པ་ གང་+ཡིན་ -པ་ དང་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ སྟོང་པ་ཉིད་ གང་+ཡིན་ -པ་ དང་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ དབེན་པ་ གང་+ཡིན་ -པ་ དང་ །  དྲན་པ་ ཉེ་བ་ #ར་ གཞག་ -པ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ གང་+ཡིན་ -པ་ དང་ །  བྱང་ཆུབ་སེམས་དཔའ་ གང་+ཡིན་ -པ་ དང་ །  སྔོན་ གྱི་ མཐའ་ གང་+ཡིན་ -པ་ དང་ །  ཕྱི་མ་ #འི་ མཐའ་ གང་+ཡིན་ -པ་ དང་ །  དབུས་ གང་+ཡིན་ -པ་ དེ་ ཐམས་ཅད་ ཀྱང་ གཉིས་ སུ་ མྱེད་ དེ་ གཉིས་ སུ་ བྱར་ མྱེད་ -དོ་ ། ། ཚེ་ -དང་ -ལྡན་པ་ ཤ་ར་དྭ་ཏི་ #འི་ བུ་ །  ཡང་དག་ -པ+ར་ སྤོང་བ་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། ཡང་དག་ -པ+ར་ སྤོང་བ་ སྟོང་པ་ཉིད་ ཀྱི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། །

 ཚེ་ -དང་ -ལྡན་པ་ ཤ་ར་དྭ་ཏི་ #འི་ བུ་ །  དེ་ལྟར་ ན་ སྙིང་རྗེ་ ཆེན་པོ་ མྱེད་ -པ་ གང་+ཡིན་ -པ་ དང་ །  སྙིང་རྗེ་ ཆེན་པོ་ སྟོང་པ་ཉིད་ གང་+ཡིན་ -པ་ དང་ །  སྙིང་རྗེ་ ཆེན་པོ་ དབེན་པ་ གང་+ཡིན་ -པ་ དང་ །  སྙིང་རྗེ་ ཆེན་པོ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ གང་+ཡིན་ -པ་ དང་ །  བྱང་ཆུབ་སེམས་དཔའ་ གང་+ཡིན་ -པ་ དང་ །  སྔོན་ གྱི་ མཐའ་ གང་+ཡིན་ -པ་ དང་ །  ཕྱི་མ་ #འི་ མཐའ་ གང་+ཡིན་ -པ་ དང་ །  དབུས་ གང་+ཡིན་ -པ་ དེ་ ཐམས་ཅད་ ཀྱང་ གཉིས་ སུ་ མྱེད་ དེ་ གཉིས་ སུ་ མྱེད་ དེ་ གཉིས་ སུ་ བྱར་ མྱེད་ -དོ་ ། ། ཚེ་ -དང་ -ལྡན་པ་ ཤ་ར་དྭ་ཏི་ #འི་ བུ་ །  སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ སྟོང་པ་ཉིད་ ཀྱི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ དབེན་པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ སྔོན་ གྱི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ །  སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ སྟོང་པ་ཉིད་ ཀྱི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མི་ དམྱིགས་ -སོ་ ། ། སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ དབེན་པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མྱི་ དམྱིགས་ -སོ་ ། ། སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ ངོ་བོ་ -ཉིད་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ ཕྱི་མ་ #འི་ མཐའ་ ￥ར་ མི་ དམྱིགས་ -སོ་ ། ། སངས་རྒྱས་ ཀྱི་ ཆོས་མ་ འདྲེས་པ་ མྱེད་ -པ་ #འི་ ཕྱིར་ །  བྱང་ཆུབ་སེམས་དཔའ་ དབུས་ སུ་ མྱི་ དམྱིགས་ -སོ་ ། །
