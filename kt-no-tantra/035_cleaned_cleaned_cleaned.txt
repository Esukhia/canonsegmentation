 རྒྱ་མཚོ་ #འི་ ལྷ་ བསམ་ གྱིས་ མི་ ཁྱབ་པ་ དག་ དང་ ཡང་ ཐབས་ གཅིག་ སྟེ་ །  འདི་ལྟར་ རྒྱ་མཚོ་ #འི་ ལྷ་ རིན་པོ་ཆེ་ #འི་ འོད་ འབྱུང་བ་ #འི་ དཔལ་ དང་ །  རྡོ་རྗེ་ ཀུན་ ནས་ སྲ་ ཞིང་ མྱི་ ཆོམས་ -པ་ #འི་ རྒྱལ་མཚན་ དང་ ལྡན་པ་ དང་ །  རྡུལ་ དང་ དྲི་མ་ མྱེད་ -པ་ #འི་ གནས་ དང་ ལྡན་པ་ དང་ །  ཀུན་ ནས་ཆུ་ དང་ རི་བོ་ དང་ ཁྱིམ་ གྱི་ དཀྱིལ་འཁོར་ དང་ །  སྐྱབས་ རིན་པོ་ཆེ་ མྱིག་ གི་ ཟླ་བ་ #འི་ དཔལ་ དང་ །  ཀླུ་ #འི་ དཔལ་ པད་མོ་ སྣ་ཚོགས་ ཀྱི་ གཙུག་ཕུད་ དང་ །  ཆུ་ ལྟ་བུ་ #ར་ -རོ་ མཐུན་པ་ #ར་ འཛིན་པ་ #འི་ འོད་ དང་ །  རིན་པོ་ཆེ་ #འི་ འོད་འཕྲོ་ -བ་ #འི་ མེ་ཏོག་ གི་ ཚོན་ གྱི་ འོད་ དང་ །  རི་རབ་ སྲ་བ་ #འི་ རྡོ་རྗེ་ སྣ་ཚོགས་ དང་ །  རྒྱ་མཚོ་ #འི་ ལྷ་ འབྲུག་སྒྲ་ སྒྲོགས་ -པ་ རྒྱ་མཚོ་ #འི་ ཤུགས་ ཀྱི་ དབྱངས་ དང་ །  དེ་དག་ ལ་སོགས་པ་ རྒྱ་མཚོ་ #འི་ ལྷ་ བསམ་ གྱིས་ མྱི་ ཁྱབ་པ་ ཀུན་ ཀྱང་ སངས་རྒྱས་ ཀྱི་ ཡོན་ཏན་ རྒྱ་མཚོ་ #འི་ མཐའ་ ￥ར་ ཕྱིན་པ་ #ར་ རྒྱས་པ་ #ར་ བྱེད་པ་ ཤ་སྟག་ དང་ ཐབས་ གཅིག་ -གོ་ ། 
 མེ་ #འི་ ལྷ་ གྲངས་མེད་ -པ་ དག་ དང་ ཡང་ ཐབས་ གཅིག་ སྟེ་ །  འདི་ལྟར་ མེ་ #འི་ ལྷ་ གཟི་བརྗིད་ འབར་ ཞིང་ འཕྲོ་བ་ #འི་ འོད་ དང་ །  ཀུན་ ཏུ་ འབར་ ཞིང་ འཕྲོ་བ་ #འི་ རྒྱལ་མཚན་ གྱི་ དཀྱིལ་འཁོར་ དང་ །  འོད་ རྒྱ་ཆེན་པོ་ #ས་ ཕྱོགས་ རྒྱས་པ་ #ར་ བྱེད་པ་ དང་ །  རྒྱལ་མཚན་ གྱི་ རྡུལ་ གྱི་ སྒོས་ རྒྱས་པ་ #ར་ བྱེད་པ་ དང་ །  གཟི་བརྗིད་ མི་ཟད་ -པ་ #འི་ གཙུག་ཕུད་ དང་ །  དཀའ་ཐུབ་ སྣ་ཚོགས་ ཀྱི་ མིག་ གི་ དཔལ་ དང་ །  ཕྱོགས་ ཁྱིམ་ དུ་ གྱུར་པ་ #འི་ རི་རབ་ ཀྱི་ སྤོ་ དང་ །  འོད་ དྲག་ཤུལ་ ཅན་ གྱི་ དབང་ཕྱུག་ དང་ །  མུན་པ་ དང་ རབ་རིབ་ སེལ་ ཅིང་ ཕྱོགས་ རྣམ་པ+ར་ སྣང་བར་ བྱེད་པ་ དང་ །  མེ་ #འི་ ལྷ་ འོད་འཕྲོ་ -བ་ #འི་ སྤྲིན་ གྱི་ དབང་ཕྱུག་ དང་ །  དེ་དག་ ལ་སོགས་པ་ མེ་ #འི་ ལྷ་ གྲངས་མེད་ -པ་ ཀུན་ ཀྱང་ སེམས་ཅན་ ཐམས་ཅད་ ཀྱི་ གདུང་བ་ སེལ་བ་ #འི་ སྣང་བ་ སྟོན་པ་ ཤ་སྟག་ དང་ ཐབས་ གཅིག་ -གོ་ ། ཆུ་ #འི་ ཕུང་པོ་ #འི་ ལྷ་ ཚད་ མེད་པ་ དག་ དང་ ཡང་ ཐབས་ གཅིག་ སྟེ་ །  འདི་ལྟར་ ཆུ་ #འི་ ཕུང་པོ་ #འི་ ལྷ་ སྤྲིན་ དང་ ན་བུན་ ཀུན་ ཏུ་ གྲུབ་པ་ #འི་ རྒྱལ་མཚན་ དང་ །  སྤྲིན་སྒྲ་ #འི་ ཤུགས་ རྒྱ་མཚོ་ #འི་ སྤྲིན་ དང་ །  ཆུ་ སྣ་ཚོགས་ ཀྱི་ གཙུག་ཕུད་ དང་ །  བཟང་པོ་ #ར་ འཁྱིལ་ -བ་ གྲུབ་པ་ #འི་ དཔལ་ དང་ །  ཀུན་ ནས་ སྤོས་དྲི་ མ་ མྱེད་ -པ་ #འི་ མྱུ་གུ་ #འི་ དཔལ་ དང་ །  དབའ་རླབས་ ཀྱི་ སྒྲ་ བསོད་ནམས་ སྟེགས་ ཀྱི་ འོད་ དང་ །  ཚིམ་པ་ #ར་ བྱ་བ་ ལ་ དབང་ བྱེད་པ་ ཡོངས་སུ་ སྒྲུབ་པ་ དང་ །  ས་+འི་ རྡུལ་ སྙེད་ ཀྱི་ སྒོ་ ནས་ དབྱངས་ འབྱུང་བ་ དང་ །  འོད་ ཀུན་ ཏུ་ འགྲོ་ ཞིང་ རྒྱས་པ་ #ར་ བྱེད་པ་ དང་ །  ཆུ་ #འི་ ཕུང་པོ་ #འི་ ལྷ་ རྒྱ་མཚོ་ #འི་ དབྱངས་ ཀྱི་ ང་རོ་ #འི་ འོད་ དང་ །  དེ་དག་ ལ་ སྩོགས་པ་ #འི་ ཆུ་ #འི་ ཕུང་པོ་ #འི་ ལྷ་ ཚད་ མེད་པ་ ཀུན་ ཀྱང་ འགྲོ་བ་ ཐམས་ཅད་ ལ་ ཕན་པ་ དང་ བདེ་བ་ #འི་ སྐྱབས་ སུ་ བརྩོན་པ་ ཤ་སྟག་ དང་ ཐབས་ གཅིག་ -གོ་ །

 ལྷ་ #འི་ རྒྱལ་པོ་ དབང་ སྒྱུར་ ལ་སོགས་པ་ བགྲང་ དུ་ མེད་པ་ དག་ དང་ ཡང་ ཐབས་ གཅིག་ སྟེ་ །  འདི་ལྟར་ ལྷ་ #འི་ རྒྱལ་པོ་ དབང་ སྒྱུར་ དང་ །  ལྷ་ #འི་ རྒྱལ་པོ་ མིག་ བཟང་པོ་ #འི་ བློ་གྲོས་ དང་ །  ཅོད་པན་ སྣ་ཚོགས་ ཀྱི་ རྒྱལ་མཚན་ དང་ །  བརྩོན་པ་ #འི་ བློ་གྲོས་ དང་ །  གོ་འཕང་ རྣམ་པ+ར་ གནོན་པ་ #འི་ སྒྲ་དབྱངས་ སྣ་ཚོགས་ དང་ །  འོད་ ཡིད་ དུ་ འོང་བ་ #འི་ རྒྱལ་མཚན་ དང་ །  སྤྱོད་ཡུལ་ ཞི་བ་ #འི་ སྒོ་ དང་ །  འཁོར་ལོ་ #འི་ རི་མོ་ བཟང་པོ་ #འི་ རྒྱལ་མཚན་ དང་ །  མེ་ཏོག་ བློ་གྲོས་ སེང་གེ་ #འི་ དབང་ཕྱུག་ དང་ །  ལྷ་ #འི་ རྒྱལ་པོ་ དབང་པོ་ #འི་ སྟོབས་ རི་མོ་ བཟང་པོ་ #འི་ འོད་ དང་ །  དེ་དག་ ལ་སོགས་པ་ དབང་ སྒྱུར་ གྱི་ ལྷ་ #འི་ རྒྱལ་པོ་ བགྲང་ དུ་ མེད་པ་ ཀུན་ ཀྱང་ །  ཆོས་ ཀྱི་ དབང་ རྒྱ་ཆེན་པོ་ ལ་ རབ་ཏུ་ བརྩོན་པ་ ཤ་སྟག་ དང་ ཐབས་ གཅིག་ -གོ་ ། བསམ་ གྱིས་ མི་ ཁྱབ་པ་ #འི་ ཚངས་པ་ ཆེན་པོ་ དག་ དང་ ཡང་ ཐབས་ གཅིག་ སྟེ་ །  འདི་ལྟར་ ཚངས་པ་ ཆེན་པོ་ རབ་ཏུ་ སྣང་བ་ དང་ །  བློ་གྲོས་ ཀྱི་ འོད་ དང་ །  བློ་གྲོས་ བཟང་པོ་ #འི་ སེམས་ སྣང་བ་ དང་ །  ཀུན་ ཏུ་ དབྱངས་ ཀྱི་ སྤྲིན་ དང་ །  འཇིག་རྟེན་ དབང་པོ་ #འི་ སྒྲ་ དང་ །  མིག་ རབ་ཏུ་ ཞི་ ཞིང་ འོད་ ཤིན་ཏུ་ ཟབ་པ་ དང་ །  ཕྱོགས་ སུ་ སྣང་བ་ འགྱེད་པ་ དང་ །  སྤྲུལ་པ་ #འི་ སྒྲ་དབྱངས་ དང་ །  མིག་ གི་ འོད་ཟེར་ མངོན་པར་ སྣང་བ་ དང་ །  ཚངས་པ་ ཆེན་པོ་ དབྱངས་ རྒྱ་མཚོ་ ཡིད་ དུ་ འོང་བ་ དང་ །  དེ་དག་ ལ་སོགས་པ་ ཚངས་པ་ ཆེན་པོ་ བསམ་ གྱིས་ མི་ ཁྱབ་པ་ ཀུན་ ཀྱང་ །  སེམས་ཅན་ ཐམས་ཅད་ ཀྱི་ སེམས་ སྣང་བར་ བྱ་བ་ དང་ །  སིམ་པ་ #ར་ བྱ་བ་ ལ་ བྱམས་པ་ ཆེན་པོ་ ལ་ གནས་པ་ ཤ་སྟག་ དང་ ཐབས་ གཅིག་ -གོ་ །
