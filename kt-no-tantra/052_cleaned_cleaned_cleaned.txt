 རིགས་ -ཀྱི་ -བུ་ ཁྱོད་ དེ་ ལ་ ཕྱག་ -འཚལ་ ཞིང་ ཐེ་ཚོམ་ དུ་ གྱུར་པ་ ཡང་ ཞུ་བ་ #འི་ དུས་ ལ་ བབ་ -བོ་ ། ། དེ་ནས་ བྱང་ཆུབ་སེམས་དཔའ་ སེམས་དཔའ་ ཆེན་པོ་ ལུས་ དཔག་ ཏུ་ མེད་པ་ ཞེས་བྱ་བ་ འདུག་པ་ ལས་ ལངས་ ནས་ སངས་རྒྱས་ ཀྱི་ ཞབས་ ལ་ སྤྱི་བོ་ #ས་ ཕྱག་ -འཚལ་ ཏེ་ །  ལན་ གསུམ་ བསྐོར་ ནས་ བྱང་ཆུབ་སེམས་དཔའ་ གྲངས་མེད་ དཔག་ ཏུ་ མེད་པ་ དང་ ཐབས་ཅིག་ ཏུ་ སངས་རྒྱས་ ཀྱི་ ཞིང་ དེ་ནས་ མི་མཇེད་ ཀྱི་ འཇིག་རྟེན་ གྱི་ ཁམས་ གང་ ན་བ་ དེར་ དོང་ ངོ་ ། ། དེ་ #འི་ -ཚེ་ ན་ སྟོང་གསུམ་ གྱི་ སྟོང་ -ཆེན་པོ་ #འི་ འཇིག་རྟེན་ གྱི་ ཁམས་ རྣམ་པ་ དྲུག་ ཏུ་ གཡོས་ ནས་ འཁོར་ གྱི་ ནང་ ན་ ཚངས་པ་ དང་ །  བརྒྱ་བྱིན་ དང་ །  ཕྱོགས་ བཞི་ #འི་ མགོན་པོ་ དང་ །  བདུད་ སྡིག་ཏོ་ ཅན་ དང་ །  དབང་ སྒྱུར་ ཆེན་པོ་ ལ་སོགས་པ་ དེ་དག་ གིས་ ས་གཡོས་ -པ་ མཐོང་ ནས་ །  ལུས་ སྐྱི་བུང་ ཞེས་ བྱེད་ དེ་ ལྕེ་ ཡང་ སྐམས་ -པ་ ཙམ་ དུ་ དངངས་སྐྲག་ པར་ གྱུར་ ཏེ་ །  སོ་སོ་ #ར་ འགྲོ་བ་ #ར་ ཆས་ ནས་ ལུས་ ཀྱི་ འོད་ དང་ གཟི་བརྗིད་ ཀྱང་ མི་སྣང་བ་ #ར་ གྱུར་ ཏོ་ ། ། དེ་ནས་ འཇམ་དཔལ་ གཞོན་ནུ་ #ར་ གྱུར་པ་ འདུག་པ་ ལས་ ལངས་ ཏེ་ །  འཁོར་ གྱི་ དཀྱིལ་འཁོར་ རྣམས་ ལ་ འདི་སྐད་ ཅེས་ སྨྲས་ -སོ་ ། ། རིགས་ -ཀྱི་ -བུ་ དག་ ཁྱེད་ འཇིགས་ པར་མ་ བྱེད་ ཅིག་ ། 
 ཁྱེད་ མ་ འཇིགས་ ཤིག་ ། དེ་ ཅིའི་ཕྱིར་ ཞེ་ན་ །  འདི་ ནས་ ཤར་ཕྱོགས་ ན་ འཇིག་རྟེན་ གྱི་ ཁམས་ ཀླུང་ གང་ གཱའི་ བྱེ་མ་ གྲངས་མེད་ དཔག་ ཏུ་ མེད་པ་ འདས་པ་ #འི་ ཕ་རོལ་ ན་ འཇིག་རྟེན་ གྱི་ ཁམས་ ཡིད་ དུ་ འོང་བ་ #འི་ སྒྲ་སྙན་ ཞེས་བྱ་བ་ ན་ སངས་རྒྱས་ ནམ་མཁའ་ དང་ མཉམ་པ་ ཞེས་བྱ་བ་ དེ་བཞིན་གཤེགས་པ་ དགྲ་བཅོམ་ -པ་ ཡང་དག་ -པར་ རྫོགས་པ་ #འི་ སངས་རྒྱས་ མཚན་ སུམ་ཅུ་རྩ་གཉིས་ དང་ ལྡན་པ་ བཞུགས་ ཏེ་ །  དེ་ ན་ བྱང་ཆུབ་སེམས་དཔའ་ ལུས་ དཔག་ ཏུ་ མེད་པ་ ཞེས་བྱ་བ་ འཁོར་ བྱང་ཆུབ་སེམས་དཔའ་ གྲངས་མེད་ -པ་ དང་ བཅས་ ཏེ་ །  འཇིག་རྟེན་ གྱི་ ཁམས་ འདིར་ དེ་བཞིན་གཤེགས་པ་ ལ་ མཆོད་ དུ་ འོང་བ་ #ས་ བྱང་ཆུབ་སེམས་དཔའ་ དེ་ #འི་ གཟི་བརྗིད་ དང་ བྱིན་ གྱིས་ ཁྱེད་ ཀྱི་ ལུས་ ཀྱི་ འོད་ ཀྱང་ མི་སྣང་བ་ #ར་ གྱུར་ ཏོ་ ། །
