 བཅོམ་ལྡན་འདས་ ལེགས་ -སོ་ ཞེས་ གསོལ་ ཏེ་ །  བྱང་ཆུབ་སེམས་དཔའ་ སེམས་དཔའ་ ཆེན་པོ་ བློ་གྲོས་ རྒྱ་མཚོ་ བཅོམ་ལྡན་འདས་ ཀྱི་ ལྟར་ མཉན་པ་ དང་ །  བཅོམ་ལྡན་འདས་ ཀྱིས་ འདི་སྐད་ ཅེས་ བཀའ་ སྩལ་ -ཏོ་ ། ། བློ་གྲོས་ རྒྱ་མཚོ་ འདི་ ལ་ བྱང་ཆུབ་སེམས་དཔའ་ སེམས་དཔའ་ ཆེན་པོ་ ནི་ དགེ་བ་ #འི་ རྩ་བ་ #ས་ བསྟན་པ་ ཡིན་ ཏེ་ །  དགེ་བ་ #འི་ བསམ་པ་ དང་ ལྡན་པ་ །  ངེས་པ་ #འི་ ཕུང་པོ་ ལ་ རབ་ཏུ་ གནས་པ་ །  ཚོགས་ བསགས་པ་ །  དབང་པོ་ རྣོ་བ་ །  དགེ་བ་ #འི་ བཤེས་གཉེན་ གྱིས་ ཡོངས་སུ་ ཟིན་པ་ །  སངས་རྒྱས་ ཀྱིས་ བྱིན་ གྱིས་ བརླབས་པ་ །  རྒྱུ་ #འི་ སྟོབས་ ཅན་ ཡིན་ ཏེ་ །  དེ་ -ནི་ སངས་རྒྱས་ བཅོམ་ལྡན་འདས་ རྣམས་ མཉེས་པ་ #ར་ བྱེད་ །  བསྟི་སྟང་ དུ་ བྱེད་ །  བླ་མ་ #ར་ བྱེད་ །  རིམ་གྲོ་ #ར་ བྱེད་ །  མཆོད་པ་ #ར་ བྱེད་ -དོ་ ། ། སངས་རྒྱས་ བཅོམ་ལྡན་འདས་ དེ་དག་ གི་ གཟུགས་ ཕུན་སུམ་ -ཚོགས་པ་ མཐོང་ ཞིང་ གསུང་སྙན་ -པ་ ཐོས་ ཏེ་ །  འཁོར་ ཕུན་སུམ་ -ཚོགས་པ་ ཡང་ མཐོང་ ཞིང་ ཆགས་པ་ མེད་པ་ #འི་ མཁྱེན་པ་ ཡང་ ཐོས་ ལ་ །  རྫུ་འཕྲུལ་ གྱི་ ཆོ་འཕྲུལ་ ལམ་ །  ཀུན་ བརྗོད་པ་ #འི་ ཆོ་འཕྲུལ་ ལམ་ །  རྗེས་+སུ་ བསྟན་པ་ #འི་ ཆོ་འཕྲུལ་ གྱིས་ སེམས་ཅན་ འདུལ་བ་ མཛད་པ་ ཡང་ མཐོང་ ནས་ སེམས་ཅན་ རྣམས་ ལ་ སྙིང་རྗེ་ ཆེན་པོ་ སྐྱེས་ ཏེ་ །  བླ་ ན་ མེད་པ་ ཡང་དག་ -པ+ར་ རྫོགས་པ་ #འི་ བྱང་ཆུབ་ ཏུ་ སེམས་བསྐྱེད་ ནས་ །  བརྩོན་འགྲུས་ བརྩམས་པ་ ཡིན་ ཏེ་ །  དགེ་བ་ #འི་ ཆོས་ ཡོངས་སུ་ ཚོལ་ བ་ལ་ བརྩོན་ ཞིང་ ཐམས་ཅད་ མཁྱེན་པ་ #འི་ ཡེ་ཤེས་ བརྗེད་པ་ མེད་པ་ #ས་ སེམས་ ཡོངས་སུ་ དག་པ་ #ར་ རྣལ་འབྱོར་ དུ་ བྱེད་ -དོ་ ། །
